		}

			check_variable = { num_owned_states > 1 }

		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = imperial_unionism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@imperial_unionism
				}
			}
			if = {
				limit = {
					has_government = eurasianism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@eurasianism
				}
			}
			if = {
				limit = {
					has_government = constitutional_monarchy
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@constitutional_monarchy
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = sultanism
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_sultanism_civil_war
			clr_country_flag = military_support_for_sultanism_civil_war
			clr_country_flag = civil_support_for_sultanism_civil_war
			set_country_flag = ideology_civil_war
		}

			factor = 0
		}

	ban_sultanism_party = {

		icon = generic_civil_support

		available = {
			sultanism < 0.2
			if = {
				limit = {
					tag = SPR
				}
				NOT = {
					has_idea = SPR_political_violence
				}
			}
		}

		visible = {
			NOT = { has_government = sultanism }
			sultanism > 0
			if = { 
				limit = { 
					has_dlc = "Battle for the Bosporus"	
				}
				NOT = { original_tag = GRE }
			}
		}

		remove_trigger = {
			hidden_trigger = {
				has_government = sultanism
			}
		}
	
		modifier = {
			sultanism_drift = -0.05
		}
	
		days_remove = -1
		cost = 100
		
		complete_effect = {
			hidden_effect = {
				set_temp_variable = {
					var = ban_ideology_var
					value = party_popularity@sultanism
				}
				multiply_temp_variable = {
					var = ban_ideology_var
					value = -1
				}
				add_popularity = {
					ideology = sultanism
					popularity = ban_ideology_var
				}
				multiply_temp_variable = {
					var = ban_ideology_var
					value = 0.66
				}
				set_country_flag = banned_sultanism_flag
			}
			remove_sultanism_spirits_effect = yes
			custom_effect_tooltip = ban_sultanism_party_tt
			add_stability = ban_ideology_var
		}

		ai_will_do = {
			factor = 0
		}
	}
